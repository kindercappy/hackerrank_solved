function super_reduced_string(s){
    // Complete this function
    var regex = /([a-z])\1/g;
    var similar = true;
    var char = "";
    while(similar){
        s = s.replace(regex,"");
        char = s.match(regex);
        if(char===null){
            similar =false;
        }
    }
    if(s==="")return "Empty String";
    return s;
}

                            var reversed = 0;

                             number /= 10;
                            }
                           
                            return reversed;
                           }
